def to_s

    # Formato para mostrar los datos flotantes del personaje
    formato='%.6f'

    return "#{@name}[i:#{format(formato,@intelligence)}, s:#{format(formato,@strength)}, "+
    "h:#{format(formato,@health)}, p:(#{@row}, #{@col})]"
end
